  38|       UnicodeString Line = FSecureShell->ReceiveLine();

  47|         /* TODO 1 : Show stderror to user? */

  50|         {
  51|           // coIgnoreWarnings should allow batch transfer to continue when
  52|           // download of one the files fails (user denies overwriting
  53|           // of target local file, no read permissions...)

  55|             coOnlyReturnCode | coIgnoreWarnings);
  56|           if (!Initialized)
  57|           {
  58|             throw Exception(L"");
  59|           }
  60|         }
  61|         catch(Exception & E)

  64|           {
  65|             FTerminal->TerminalError(&E, LoadStr(SCP_INIT_ERROR));

  84|             // Error (already logged by ReceiveLine())
  85|             throw EScpFileSkipped(NULL, FMTLOAD(REMOTE_ERROR, (Line)));

  87|           case 2:
  88|             // Fatal error, terminate copying
  89|             FTerminal->TerminalError(Line);
  90|             return; // Unreachable
  91| 
  92|           case L'E': // Exit
  93|             FSecureShell->SendNull();
  94|             return;
  95| 
  96|           case L'T':
  97|             unsigned long MTime, ATime;
  98|             if (swscanf(Line.c_str(), L"%ld %*d %ld %*d",  &MTime, &ATime) == 2)
  99|             {
 100|               FileData.Modification = UnixToDateTime(MTime, FTerminal->SessionData->DSTMode);
 101|               FSecureShell->SendNull();

 103|               FileData.SetTime = 2;
 104|               continue;
 105|             }
 106|               else
 107|             {
 108|               SCPError(LoadStr(SCP_ILLEGAL_TIME_FORMAT), False);
 109|             }
 110|

 112|           case L'D':

 123|         try

 129|           // Security fix: ensure the file ends up where we asked for it.
 130|           // (accept only filename, not path)
 131|           UnicodeString OnlyFileName = UnixExtractFileName(Line);
 132|           if (Line != OnlyFileName)
 133|           {
 134|             FTerminal->LogEvent(FORMAT(L"Warning: Remote host set a compound pathname '%s'", (Line)));
 135|           }
 136| 
 137|           FullFileName = SourceDir + OnlyFileName;
 138|           OperationProgress->SetFile(FullFileName);
 139|           OperationProgress->SetTransferSize(TSize);
 140|         }
 141|         catch (Exception &E)
 142|         {
 143|           {
 144|             TSuspendFileOperationProgress Suspend(OperationProgress);
 145|             FTerminal->Log->AddException(&E);
 146|           }
 147|           SCPError(LoadStr(SCP_ILLEGAL_FILE_DESCRIPTOR), false);
 148|         }
 149|

 152|         {

 169|           SCPError(L"", false);

 171|         }

 176|           FTerminal->ChangeFileName(
 177|             CopyParam, OperationProgress->FileName, osRemote,
 178|             Level == 0);
 179|         UnicodeString DestFileName =
 180|           IncludeTrailingBackslash(TargetDir) + DestFileNameOnly;
 181| 
 182|         FileData.Attrs = FileGetAttrFix(ApiPath(DestFileName));
 183|         // If getting attrs fails, we suppose, that file/folder doesn't exists
 184|         FileData.Exists = (FileData.Attrs != -1);
 185|         if (Dir)
 186|         {
 187|           if (FileData.Exists && !(FileData.Attrs & faDirectory))
 188|           {
 189|             SCPError(FMTLOAD(NOT_DIRECTORY_ERROR, (DestFileName)), false);
