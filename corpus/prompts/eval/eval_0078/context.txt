   2| file_copy_fallback (GFile                  *source,
   3|                     GFile                  *destination,
   4|                     GFileCopyFlags          flags,
   5|                     GCancellable           *cancellable,
   6|                     GFileProgressCallback   progress_callback,
   7|                     gpointer                progress_callback_data,
   8|                     GError                **error)
   9| {
  10|   gboolean ret = FALSE;
  11|   GFileInputStream *file_in = NULL;
  12|   GInputStream *in = NULL;
  13|   GOutputStream *out = NULL;
  14|   GFileInfo *info = NULL;
  15|   const char *target;
  16|   char *attrs_to_read;
  17|   gboolean do_set_attributes = FALSE;
  18| 
  19|   /* need to know the file type */
  20|   info = g_file_query_info (source,
  21|                             G_FILE_ATTRIBUTE_STANDARD_TYPE "," G_FILE_ATTRIBUTE_STANDARD_SYMLINK_TARGET,
  22|                             G_FILE_QUERY_INFO_NOFOLLOW_SYMLINKS,
  23|                             cancellable,
  24|                             error);
  25|   if (!info)
  26|     goto out;
  27| 
  28|   /* Maybe copy the symlink? */
  29|   if ((flags & G_FILE_COPY_NOFOLLOW_SYMLINKS) &&
  30|       g_file_info_get_file_type (info) == G_FILE_TYPE_SYMBOLIC_LINK)
  31|     {
  32|       target = g_file_info_get_symlink_target (info);
  33|       if (target)
  34|         {
  35|           if (!copy_symlink (destination, flags, cancellable, target, error))
  36|             goto out;
  37| 
  38|           ret = TRUE;
  39|           goto out;
  40|         }
  41|         /* ... else fall back on a regular file copy */
  42|     }
  43|   /* Handle "special" files (pipes, device nodes, ...)? */
  44|   else if (g_file_info_get_file_type (info) == G_FILE_TYPE_SPECIAL)
  45|     {
  46|       /* FIXME: could try to recreate device nodes and others? */
  47|       g_set_error_literal (error, G_IO_ERROR, G_IO_ERROR_NOT_SUPPORTED,
  48|                            _("Can’t copy special file"));
  49|       goto out;
  50|     }
  51| 
  52|   /* Everything else should just fall back on a regular copy. */
  53| 
  54|   file_in = open_source_for_copy (source, destination, flags, cancellable, error);
  55|   if (!file_in)
  56|     goto out;
  57|   in = G_INPUT_STREAM (file_in);
  58| 
  59|   if (!build_attribute_list_for_copy (destination, flags, &attrs_to_read,
  60|                                       cancellable, error))
  61|     goto out;
  62| 
  63|   if (attrs_to_read != NULL)
  64|     {
  65|       GError *tmp_error = NULL;
  66| 
  67|       /* Ok, ditch the previous lightweight info (on Unix we just
  68|        * called lstat()); at this point we gather all the information
  69|        * we need about the source from the opened file descriptor.
  70|        */
  71|       g_object_unref (info);
  72| 
  73|       info = g_file_input_stream_query_info (file_in, attrs_to_read,
  74|                                              cancellable, &tmp_error);
  75|       if (!info)
  76|         {
  77|           /* Not all gvfs backends implement query_info_on_read(), we
  78|            * can just fall back to the pathname again.
  79|            * https://bugzilla.gnome.org/706254
  80|            */
  81|           if (g_error_matches (tmp_error, G_IO_ERROR, G_IO_ERROR_NOT_SUPPORTED))
  82|             {
  83|               g_clear_error (&tmp_error);
  84|               info = g_file_query_info (source, attrs_to_read, G_FILE_QUERY_INFO_NOFOLLOW_SYMLINKS,
  85|                                         cancellable, error);
  86|             }
  87|           else
  88|             {
  89|               g_free (attrs_to_read);
  90|               g_propagate_error (error, tmp_error);
  91|               goto out;
  92|             }
  93|         }
  94|       g_free (attrs_to_read);
  95|       if (!info)
  96|         goto out;
  97| 
  98|       do_set_attributes = TRUE;
  99|     }
 100| 
 101|   /* In the local file path, we pass down the source info which
 102|    * includes things like unix::mode, to ensure that the target file
 103|    * is not created with different permissions from the source file.
 104|    *
 105|    * If a future API like g_file_replace_with_info() is added, switch
 106|    * this code to use that.
 107|    */
 108|   if (G_IS_LOCAL_FILE (destination))
 109|     {
 110|       if (flags & G_FILE_COPY_OVERWRITE)
 111|         out = (GOutputStream*)_g_local_file_output_stream_replace (_g_local_file_get_filename (G_LOCAL_FILE (destination)),
 112|                                                                    FALSE, NULL,
 113|                                                                    flags & G_FILE_COPY_BACKUP,
 114|                                                                    G_FILE_CREATE_REPLACE_DESTINATION |
 115|                                                                    G_FILE_CREATE_PRIVATE, info,
 116|                                                                    cancellable, error);
 117|       else
 118|         out = (GOutputStream*)_g_local_file_output_stream_create (_g_local_file_get_filename (G_LOCAL_FILE (destination)),
 119|                                                                   FALSE, G_FILE_CREATE_PRIVATE, info,
 120|                                                                   cancellable, error);
 121|     }
 122|   else if (flags & G_FILE_COPY_OVERWRITE)
 123|     {
 124|       out = (GOutputStream *)g_file_replace (destination,
 125|                                              NULL,
 126|                                              flags & G_FILE_COPY_BACKUP,
 127|                                              G_FILE_CREATE_REPLACE_DESTINATION |
 128|                                              G_FILE_CREATE_PRIVATE,
 129|                                              cancellable, error);
 130|     }
 131|   else
 132|     {
 133|       out = (GOutputStream *)g_file_create (destination, G_FILE_CREATE_PRIVATE, cancellable, error);
 134|     }
 135| 
 136|   if (!out)
 137|     goto out;
 138| 
 139| #ifdef __linux__
 140|   if (G_IS_FILE_DESCRIPTOR_BASED (in) && G_IS_FILE_DESCRIPTOR_BASED (out))
 141|     {
 142|       GError *reflink_err = NULL;
 143| 
 144|       if (!btrfs_reflink_with_progress (in, out, info, cancellable,
 145|                                         progress_callback, progress_callback_data,
 146|                                         &reflink_err))
 147|         {
 148|           if (g_error_matches (reflink_err, G_IO_ERROR, G_IO_ERROR_NOT_SUPPORTED))
 149|             {
 150|               g_clear_error (&reflink_err);
 151|             }
 152|           else
 153|             {
 154|               g_propagate_error (error, reflink_err);
 155|               goto out;
 156|             }
 157|         }
 158|       else
 159|         {
 160|           ret = TRUE;
 161|           goto out;
 162|         }
 163|     }
 164| #endif
 165| 
 166| #ifdef HAVE_SPLICE
 167|   if (G_IS_FILE_DESCRIPTOR_BASED (in) && G_IS_FILE_DESCRIPTOR_BASED (out))
 168|     {
 169|       GError *splice_err = NULL;
 170| 
 171|       if (!splice_stream_with_progress (in, out, cancellable,
 172|                                         progress_callback, progress_callback_data,
 173|                                         &splice_err))
 174|         {
 175|           if (g_error_matches (splice_err, G_IO_ERROR, G_IO_ERROR_NOT_SUPPORTED))
 176|             {
 177|               g_clear_error (&splice_err);
 178|             }
 179|           else
 180|             {
 181|               g_propagate_error (error, splice_err);
 182|               goto out;
 183|             }
 184|         }
 185|       else
 186|         {
 187|           ret = TRUE;
 188|           goto out;
 189|         }
 190|     }
 191| 
 192| #endif
 193| 
 194|   /* A plain read/write loop */
 195|   if (!copy_stream_with_progress (in, out, source, cancellable,
 196|                                   progress_callback, progress_callback_data,
 197|                                   error))
 198|     goto out;
 199| 
 200|   ret = TRUE;
 201|  out:
 202|   if (in)
 203|     {
 204|       /* Don't care about errors in source here */
 205|       (void) g_input_stream_close (in, cancellable, NULL);
 206|       g_object_unref (in);
 207|     }
 208| 
 209|   if (out)
 210|     {
 211|       /* But write errors on close are bad! */
 212|       if (!g_output_stream_close (out, cancellable, ret ? error : NULL))
 213|         ret = FALSE;
 214|       g_object_unref (out);
 215|     }
 216| 
 217|   /* Ignore errors here. Failure to copy metadata is not a hard error */
 218|   /* TODO: set these attributes /before/ we do the rename() on Unix */
 219|   if (ret && do_set_attributes)
 220|     {
 221|       g_file_set_attributes_from_info (destination,
 222|                                        info,
 223|                                        G_FILE_QUERY_INFO_NOFOLLOW_SYMLINKS,
 224|                                        cancellable,
 225|                                        NULL);
 226|     }
 227| 
 228|   g_clear_object (&info);
 229| 
 230|   return ret;
 231| }
