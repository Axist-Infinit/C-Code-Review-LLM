 553| 			else
 554| #endif

 564| 			    break;	// failed to jump to file

 566| 			// want that here
 567| 			curwin->w_cursor.lnum = files[depth].lnum;
 568| 		    }
 569| 		}
 570| 		if (action != ACTION_SHOW)

 572| 		    curwin->w_cursor.col = (colnr_T)(startp - line);

 606| 	    ins_compl_check_keys(30, FALSE);
 607| 	if (got_int || ins_compl_interrupted())
 608| 	    break;

 611| 	 * Read the next line.  When reading an included file and encountering
 612| 	 * end-of-file, close the file and continue in the file that included
 613| 	 * it.

 616| 		&& vim_fgets(line = file_line, LSIZE, files[depth].fp))
 617| 	{

 656|     vim_free(files);

 667|     }

 673| 	    emsg(_(e_couldnt_find_definition));
