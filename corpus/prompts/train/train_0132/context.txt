  23| 
  24|   // The last dimension gets replaced by the embedding.
  25|   const int output_rank = (lookup_rank - 1) + (embedding_rank - 1);

  27|   // Make sure that the actual dense shape of the sparse tensor represented by

  34|   int k = 0;
  35|   size_t embedding_size = 1;
  36|   size_t lookup_size = 1;
  37|   for (int i = 0; i < lookup_rank - 1; i++, k++) {
  38|     const size_t dim = dense_shape->data.i32[i];

  60|   // Makes sure reallocation was successful.
