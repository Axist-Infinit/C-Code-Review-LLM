   3|   XCFLayerInfo *inLayerInfo,ExceptionInfo *exception)
   4| {
   5|   int
   6|     destLeft = 0,
   7|     destTop = 0;
   8| 
   9|   Image*
  10|     tile_image;
  11| 
  12|   MagickBooleanType
  13|     status;
  14| 
  15|   MagickOffsetType
  16|     saved_pos,
  17|     offset,
  18|     offset2;
  19| 
  20|   register ssize_t
  21|     i;
  22| 
  23|   size_t
  24|     width,
  25|     height,
  26|     ntiles,
  27|     ntile_rows,
  28|     ntile_cols,
  29|     tile_image_width,
  30|     tile_image_height;
  31| 
  32|   /* start reading the data */
  33|   width=ReadBlobMSBLong(image);
  34|   height=ReadBlobMSBLong(image);
  35| 
  36|   /* read in the first tile offset.
  37|    *  if it is '0', then this tile level is empty
  38|    *  and we can simply return.
  39|    */
  40|   offset=(MagickOffsetType) ReadBlobMSBLong(image);
  41|   if (offset == 0)
  42|     return(MagickTrue);
  43|   /* Initialise the reference for the in-memory tile-compression
  44|    */
  45|   ntile_rows=(height+TILE_HEIGHT-1)/TILE_HEIGHT;
  46|   ntile_cols=(width+TILE_WIDTH-1)/TILE_WIDTH;
  47|   ntiles=ntile_rows*ntile_cols;
  48|   for (i = 0; i < (ssize_t) ntiles; i++)
  49|   {
  50|     status=MagickFalse;
  51|     if (offset == 0)
  52|       ThrowBinaryException(CorruptImageError,"NotEnoughTiles",image->filename);
  53|     /* save the current position as it is where the
  54|      *  next tile offset is stored.
  55|      */
  56|     saved_pos=TellBlob(image);
  57|     /* read in the offset of the next tile so we can calculate the amount
  58|        of data needed for this tile*/
  59|     offset2=(MagickOffsetType)ReadBlobMSBLong(image);
  60|     /* if the offset is 0 then we need to read in the maximum possible
  61|        allowing for negative compression */
  62|     if (offset2 == 0)
  63|       offset2=(MagickOffsetType) (offset + TILE_WIDTH * TILE_WIDTH * 4* 1.5);
  64|     /* seek to the tile offset */
  65|     if (offset2 > GetBlobSize(image))
  66|       ThrowBinaryException(CorruptImageError,"InsufficientImageDataInFile",
  67|         image->filename);
  68|     if (SeekBlob(image, offset, SEEK_SET) != offset)
  69|       ThrowBinaryException(CorruptImageError,"InsufficientImageDataInFile",
  70|         image->filename);
  71| 
  72|       /*
  73|         Allocate the image for the tile.  NOTE: the last tile in a row or
  74|         column may not be a full tile!
  75|       */
  76|       tile_image_width=(size_t) (destLeft == (int) ntile_cols-1 ?
  77|         (int) width % TILE_WIDTH : TILE_WIDTH);
  78|       if (tile_image_width == 0)
  79|         tile_image_width=TILE_WIDTH;
  80|       tile_image_height = (size_t) (destTop == (int) ntile_rows-1 ?
  81|         (int) height % TILE_HEIGHT : TILE_HEIGHT);
  82|       if (tile_image_height == 0)
  83|         tile_image_height=TILE_HEIGHT;
  84|       tile_image=CloneImage(inLayerInfo->image,tile_image_width,
  85|         tile_image_height,MagickTrue,exception);
  86| 
  87|       /* read in the tile */
  88|       switch (inDocInfo->compression)
  89|       {
  90|         case COMPRESS_NONE:
  91|           if (load_tile(image,tile_image,inDocInfo,inLayerInfo,(size_t) (offset2-offset),exception) == 0)
  92|             status=MagickTrue;
  93|           break;
  94|         case COMPRESS_RLE:
  95|           if (load_tile_rle (image,tile_image,inDocInfo,inLayerInfo,
  96|               (int) (offset2-offset),exception) == 0)
  97|             status=MagickTrue;
  98|           break;
  99|         case COMPRESS_ZLIB:
 100|           ThrowBinaryException(CoderError,"ZipCompressNotSupported",
 101|             image->filename)
 102|         case COMPRESS_FRACTAL:
 103|           ThrowBinaryException(CoderError,"FractalCompressNotSupported",
 104|             image->filename)
 105|       }
 106| 
 107|       /* composite the tile onto the layer's image, and then destroy it */
 108|       (void) CompositeImage(inLayerInfo->image,tile_image,CopyCompositeOp,
 109|         MagickTrue,destLeft * TILE_WIDTH,destTop*TILE_HEIGHT,exception);
 110|       tile_image=DestroyImage(tile_image);
 111| 
 112|       /* adjust tile position */
 113|       destLeft++;
 114|       if (destLeft >= (int) ntile_cols)
 115|         {
 116|           destLeft = 0;
 117|           destTop++;
 118|         }
 119|       if (status != MagickFalse)
 120|         return(MagickFalse);
 121|       /* restore the saved position so we'll be ready to
 122|        *  read the next offset.
 123|        */
 124|       offset=SeekBlob(image, saved_pos, SEEK_SET);
 125|       /* read in the offset of the next tile */
 126|       offset=(MagickOffsetType) ReadBlobMSBLong(image);
 127|     }
 128|   if (offset != 0)
 129|     ThrowBinaryException(CorruptImageError,"CorruptImage",image->filename)
 130|   return(MagickTrue);
 131| }
