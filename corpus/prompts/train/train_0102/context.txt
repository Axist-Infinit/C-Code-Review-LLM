   2| llvm::Optional<Value> simplifyBroadcast(ShapeComponentAnalysis& analysis,
   3|                                         ValueRange shapes, Location loc,
   4|                                         OpBuilder* builder) {
   5|   // First find the input shape with the largest rank.
   6|   SmallVector<ArrayRef<ShapeComponentAnalysis::SymbolicExpr>> shapes_found;
   7|   size_t maxRank = 0;
   8|   for (const auto &shape : llvm::enumerate(shapes)) {
   9|     auto found_shape = analysis.GetValueInfo(shape.value());
  10|     if (!found_shape) return {};
  11|     shapes_found.push_back(*found_shape);
  12|     maxRank = std::max(maxRank, found_shape->size());
  13|   }
  14|   if (maxRank == 0) {
  15|     return Value(builder->create<tensor::FromElementsOp>(
  16|         loc, shapes[0].getType(), SmallVector<Value>()));
  17|   }
  18| 
  19|   SmallVector<const ShapeComponentAnalysis::SymbolicExpr*> joined_dimensions(
  20|       maxRank);
  21|   SmallVector<std::pair<Value, int64_t>> shape_and_rank_for_dim(maxRank);
  22|   for (const auto &shape : llvm::enumerate(shapes_found)) {
  23|     for (const auto &dim : llvm::enumerate(llvm::reverse(shape.value()))) {
  24|       // 1 dimensions don't contribute to the final result.
  25|       if (dim.value().isConstant(1)) continue;
  26|       // If it's not a 1 dimension it will be present in the result. Remember
  27|       // where it came from.
  28|       auto index = maxRank - dim.index() - 1;
  29|       if (!joined_dimensions[index]) {
  30|         joined_dimensions[index] = &dim.value();
  31|         shape_and_rank_for_dim[index] =
  32|             std::make_pair(shapes[shape.index()], shape.value().size());
  33|         continue;
  34|       }
  35|       // Bail if the dimensions are neither equal nor 1.
  36|       if (*joined_dimensions[index] != dim.value()) return {};
  37|     }
  38|   }
  39|   // If the output is the same as one of the inputs just return that.
  40|   if (llvm::is_splat(shape_and_rank_for_dim) &&
  41|       shape_and_rank_for_dim[0].first) {
  42|     return shape_and_rank_for_dim[0].first;
  43|   }
  44|   // Otherwise rematerialize the shape from the pieces we have.
  45|   SmallVector<Value> elements;
  46|   for (int i = 0; i != maxRank; ++i) {
  47|     // 1 dimensions are filtered above, recreate the constant.
  48|     if (!shape_and_rank_for_dim[i].first) {
  49|       auto one = builder->getIntegerAttr(
  50|           shapes[0].getType().cast<RankedTensorType>().getElementType(), 1);
  51|       elements.push_back(builder->create<ConstantOp>(loc, one));
  52|       continue;
  53|     }
  54|     // Extract from one of the shapes, accounting for the reverse indexing
  55|     // performed by broadcast.
  56|     Value index = builder->create<ConstantIndexOp>(
  57|         loc, i - maxRank + shape_and_rank_for_dim[i].second);
  58|     elements.push_back(builder->create<tensor::ExtractOp>(
  59|         loc, shape_and_rank_for_dim[i].first, index));
  60|   }
  61|   return Value(builder->create<tensor::FromElementsOp>(loc, elements));
  62| }
