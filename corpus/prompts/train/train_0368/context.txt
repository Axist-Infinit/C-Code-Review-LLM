  10| 	if (mode[0] == 'r')
  11| 		return fopen(path, mode);
  12| 
  13| 	if ((mode[0] != 'a' && mode[0] != 'w') ||
  14| 	    (mode[1] &&
  15| 	     (mode[1] != '+' || mode[2]))) {
  16| 		errno = EINVAL;
  17| 		return NULL;
  18| 	}
  19| 
  20| 	if (mode[0] == 'w') {
  21| 		/* If we truncate an existing file, any non-privileged user who already has the file
  22| 		 * open would be able to read what we write, even though the file access mode is changed.
  23| 		 *
  24| 		 * If we unlink an existing file and the desired file is subsequently created via open,
  25| 		 * it leaves a window for someone else to create the same file between the unlink and the open.
  26| 		 *
  27| 		 * The solution is to create a temporary file that we will rename to the desired file name.
  28| 		 * Since the temporary file is created owned by root with the only file access persmissions being
  29| 		 * owner read and write, no non root user will have access to the file. Further, the rename to
  30| 		 * the requested filename is atomic, and so there is no window when someone else could create
  31| 		 * another file of the same name.
  32| 		 */
  33| 		fd = mkostemp(file_tmp_name, O_CLOEXEC);
  34| 	} else {
  35| 		/* Only allow append mode if debugging features requiring append are enabled. Since we
  36| 		 * can't unlink the file, there may be a non privileged user who already has the file open
  37| 		 * for read (e.g. tail -f). If these debug option aren't enabled, there is no potential
  38| 		 * security risk in that respect. */
  39| #ifndef ENABLE_LOG_FILE_APPEND
  40| 		log_message(LOG_INFO, "BUG - shouldn't be opening file for append with current build options");
  41| 		errno = EINVAL;
  42| 		return NULL;
  43| #else
  44| 		flags = O_NOFOLLOW | O_CREAT | O_CLOEXEC | O_APPEND;
  45| 
  46| 		if (mode[1])
  47| 			flags |= O_RDWR;
  48| 		else
  49| 			flags |= O_WRONLY;
  50| 
  51| 		fd = open(path, flags, S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP | S_IROTH | S_IWOTH);
  52| #endif
  53| 	}
  54| 
  55| 	if (fd == -1) {

  58| 		errno = sav_errno;
  59| 		return NULL;
  60| 	}
  61| 
  62| #ifndef ENABLE_LOG_FILE_APPEND

  90| 		}
  91| 	}

  98| 		errno = sav_errno;
