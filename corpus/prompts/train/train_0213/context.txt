   2| bool IsConstantFoldable(
   3|     const Node* n,
   4|     const std::unordered_map<string, std::vector<PartialTensorShape>>*
   5|         shape_map,
   6|     const std::function<bool(const Node*)>& consider,
   7|     int64_t max_constant_size_in_bytes,
   8|     std::unordered_map<const Node*, std::vector<Tensor>>*
   9|         shape_replacement_map) {
  10|   if (n->IsConstant()) {
  11|     // Skip constant folding resources as they cannot be deep copied.
  12|     return n->output_type(0) != DT_RESOURCE;
  13|   }
  14|   if (MaybeReplaceShapeOp(n, shape_map, shape_replacement_map)) {
  15|     return true;
  16|   }
  17|   if (n->op_def().is_stateful()) {
  18|     return false;
  19|   }
  20|   if (consider && !consider(n)) {
  21|     return false;
  22|   }
  23|   if (shape_map != nullptr) {
  24|     // We can skip the node if an output is known to be oversized.
  25|     auto shape_it = shape_map->find(n->name());
  26|     if (shape_it != shape_map->end()) {
  27|       for (int64_t i = 0; i < shape_it->second.size(); ++i) {
  28|         const auto& out_shape = shape_it->second[i];
  29|         if (out_shape.IsFullyDefined() &&
  30|             out_shape.num_elements() * DataTypeSize(n->output_type(i)) >
  31|                 max_constant_size_in_bytes) {
  32|           return false;
  33|         }
  34|       }
  35|     }
  36|   }
  37|   if (n->IsControlFlow() || n->IsSend() || n->IsRecv()) {
  38|     return false;
  39|   }
  40|   // TODO(yuanbyu): For now disable these session handle operations.
  41|   if (n->IsGetSessionHandle() || n->IsGetSessionTensor() ||
  42|       n->IsDeleteSessionTensor()) {
  43|     return false;
  44|   }
  45|   if (n->IsSource()) {
  46|     return false;
  47|   }
  48|   if (n->IsSink()) {
  49|     return false;
  50|   }
  51|   if (n->IsFakeParam()) {
  52|     return false;
  53|   }
  54|   // Since constant-folding runs on the CPU, do not attempt to constant-fold
  55|   // operators that have no CPU kernel. Also implies that we will not
  56|   // constant-fold functions.
  57|   // TODO(phawkins): allow constant-folding for functions; functions may
  58|   // be arbitrarily expensive to execute.
  59|   if (!KernelDefAvailable(DeviceType(DEVICE_CPU), n->def())) {
  60|     return false;
  61|   }
  62|   // Do not constant fold nodes which will be allocated by ScopedAllocator.
  63|   // This is because the constant-folding graph will not contain the
  64|   // `_ScopedAllocator` node, and that is necessary to be able to run a node
  65|   // that will use this allocator.
  66|   if (n->attrs().Find(kScopedAllocatorAttrName) != nullptr) {
  67|     VLOG(2) << "Skip node [" << n->DebugString()
  68|             << "] for constant folding due to scoped allocator";
  69|     return false;
  70|   }
  71|   return true;
  72| }
