   2| OFCondition WlmActivityManager::StartProvidingService()
   3| // Date         : December 10, 2001
   4| // Author       : Thomas Wilkens
   5| // Task         : Starts providing the implemented service for calling SCUs.
   6| //                After having created an instance of this class, this function
   7| //                shall be called from main.
   8| // Parameters   : none.
   9| // Return Value : Return value that is supposed to be returned from main().
  10| {
  11|   OFCondition cond = EC_Normal;
  12|   T_ASC_Network *net = NULL;
  13| 
  14|   // Make sure data dictionary is loaded.
  15|   if( !dcmDataDict.isDictionaryLoaded() )
  16|   {
  17|     DCMWLM_WARN("no data dictionary loaded, check environment variable: " << DCM_DICT_ENVIRONMENT_VARIABLE);
  18|   }
  19| 
  20| #ifndef DISABLE_PORT_PERMISSION_CHECK
  21| #ifdef HAVE_GETEUID
  22|   // If port is privileged we must be as well.
  23|   if( opt_port < 1024 && geteuid() != 0 )
  24|     return( WLM_EC_InsufficientPortPrivileges );
  25| #endif
  26| #endif
  27| 
  28| #ifdef _WIN32
  29|   /* if this process was started by CreateProcess, opt_forkedChild is set */
  30|   if (opt_forkedChild)
  31|   {
  32|     /* tell dcmnet DUL about child process status, too */
  33|     DUL_markProcessAsForkedChild();
  34| 
  35|     char buf[256];
  36|     DWORD bytesRead = 0;
  37|     HANDLE hStdIn = GetStdHandle(STD_INPUT_HANDLE);
  38| 
  39|     // read socket handle number from stdin, i.e. the anonymous pipe
  40|     // to which our parent process has written the handle number.
  41|     if (ReadFile(hStdIn, buf, sizeof(buf), &bytesRead, NULL))
  42|     {
  43|       // make sure buffer is zero terminated
  44|       buf[bytesRead] = '\0';
  45|         dcmExternalSocketHandle.set(atoi(buf));
  46|     }
  47|     else
  48|     {
  49|       DCMWLM_ERROR("cannot read socket handle: " << GetLastError());
  50|       exit(0);
  51|     }
  52|   }
  53|   else
  54|   {
  55|     // parent process
  56|     if (!opt_singleProcess)
  57|       DUL_requestForkOnTransportConnectionReceipt(cmd_argc, cmd_argv);
  58|   }
  59| #endif
  60| 
  61|   // Initialize network, i.e. create an instance of T_ASC_Network*.
  62|   cond = ASC_initializeNetwork( NET_ACCEPTOR, OFstatic_cast(int, opt_port), opt_acse_timeout, &net );
  63|   if( cond.bad() ) return( WLM_EC_InitializationOfNetworkConnectionFailed );
  64| 
  65| #if defined(HAVE_SETUID) && defined(HAVE_GETUID)
  66|   // Return to normal uid so that we can't do too much damage in case
  67|   // things go very wrong. Only works if the program is setuid root,
  68|   // and run by another user. Running as root user may be
  69|   // potentially disasterous if this program screws up badly.
  70|   if ((setuid(getuid()) == -1) && (errno == EAGAIN))
  71|   {
  72|       DCMWLM_ERROR("setuid() failed, maximum number of processes/threads for uid already running.");
  73|       return WLM_EC_InitializationOfNetworkConnectionFailed;
  74|   }
  75| #endif
  76| 
  77|   // If we get to this point, the entire initialization process has been completed
  78|   // successfully. Now, we want to start handling all incoming requests. Since
  79|   // this activity is supposed to represent a server process, we do not want to
  80|   // terminate this activity. Hence, create an endless while-loop.
  81|   while( cond.good() )
  82|   {
  83|     // Wait for an association and handle the requests of
  84|     // the calling applications correspondingly.
  85|     cond = WaitForAssociation( net );
  86| 
  87|     // Clean up any child processes if the execution is not limited to a single process.
  88|     // (On windows platform, childs are not handled via the process table,
  89|     // so there's no need to clean up children)
  90| #ifdef HAVE_FORK
  91|     if( !opt_singleProcess )
  92|       CleanChildren();
  93| #elif defined(_WIN32)
  94|     // if running in multi-process mode, always terminate child after one association
  95|     // for unix, this is done in WaitForAssociation() with exit()
  96|     if (DUL_processIsForkedChild()) break;
  97| #endif
  98|   }
  99|   // Drop the network, i.e. free memory of T_ASC_Network* structure. This call
 100|   // is the counterpart of ASC_initializeNetwork(...) which was called above.
 101|   cond = ASC_dropNetwork( &net );
 102|   if( cond.bad() ) return( WLM_EC_TerminationOfNetworkConnectionFailed );
 103| 
 104|   // return ok
 105|   return( EC_Normal );
 106| }
