   6|     /*
   7|      * Create a new pid namespace for *child* processes.  We'll have to
   8|      * fork in order to enter the new pid namespace.  A new mount namespace
   9|      * is also needed so that we can remount /proc for the new pid
  10|      * namespace.
  11|      *
  12|      * Our UNIX domain sockets have been created.  Now we can move to
  13|      * an empty network namespace to prevent TCP/IP and other network
  14|      * activity in case this process is compromised.
  15|      */
  16|     if (unshare(CLONE_NEWPID | CLONE_NEWNS | CLONE_NEWNET) != 0) {
  17|         fuse_log(FUSE_LOG_ERR, "unshare(CLONE_NEWPID | CLONE_NEWNS): %m\n");
  18|         exit(1);
  19|     }
  20| 
  21|     child = fork();
  22|     if (child < 0) {
  23|         fuse_log(FUSE_LOG_ERR, "fork() failed: %m\n");
  24|         exit(1);
  25|     }
  26|     if (child > 0) {
  27|         pid_t waited;
  28|         int wstatus;
  29| 
  30|         setup_wait_parent_capabilities();
  31| 
  32|         /* The parent waits for the child */
  33|         do {
  34|             waited = waitpid(child, &wstatus, 0);
  35|         } while (waited < 0 && errno == EINTR && !se->exited);
  36| 
  37|         /* We were terminated by a signal, see fuse_signals.c */
  38|         if (se->exited) {
  39|             exit(0);
  40|         }
  41| 
  42|         if (WIFEXITED(wstatus)) {
  43|             exit(WEXITSTATUS(wstatus));
  44|         }
  45| 
  46|         exit(1);
  47|     }
  48| 
  49|     /* Send us SIGTERM when the parent thread terminates, see prctl(2) */
  50|     prctl(PR_SET_PDEATHSIG, SIGTERM);

  52|     /*
  53|      * If the mounts have shared propagation then we want to opt out so our
  54|      * mount changes don't affect the parent mount namespace.
  55|      */
  56|     if (mount(NULL, "/", NULL, MS_REC | MS_SLAVE, NULL) < 0) {
  57|         fuse_log(FUSE_LOG_ERR, "mount(/, MS_REC|MS_SLAVE): %m\n");
  58|         exit(1);
  59|     }
