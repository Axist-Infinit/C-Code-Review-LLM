 546| 			result = dns_rdata_tostruct(&dnskeyrr, &dnskey, NULL);
 547| 			RUNTIME_CHECK(result == ISC_R_SUCCESS);
 548| 			trust_key(zone, keyname, &dnskey, false, mctx);
 549| 		}
 550| 
 551| 		if (secure && !deletekey) {
 552| 			INSIST(newkey || updatekey);
 553| 			set_refreshkeytimer(zone, &keydata, now, false);
 554| 		}
 555| 	}
 556| 
 557| 	/*
 558| 	 * RFC5011 says, "A trust point that has all of its trust anchors
 559| 	 * revoked is considered deleted and is treated as if the trust
 560| 	 * point was never configured."  But if someone revoked their
 561| 	 * active key before the standby was trusted, that would mean the
 562| 	 * zone would suddenly be nonsecured.  We avoid this by checking to
 563| 	 * see if there's pending keydata.  If so, we put a null key in
 564| 	 * the security roots; then all queries to the zone will fail.
 565| 	 */
 566| 	if (pending != 0) {
 567| 		fail_secure(zone, keyname);
 568| 	}
 569| 
 570|  done:
 571| 	if (!ISC_LIST_EMPTY(diff.tuples)) {
 572| 		/* Write changes to journal file. */
 573| 		CHECK(update_soa_serial(kfetch->db, ver, &diff, mctx,
 574| 					zone->updatemethod));
 575| 		CHECK(zone_journal(zone, &diff, NULL, "keyfetch_done"));
 576| 		commit = true;
 577| 
 578| 		DNS_ZONE_SETFLAG(zone, DNS_ZONEFLG_LOADED);
 579| 		zone_needdump(zone, 30);
 580| 	} else if (result == ISC_R_NOMORE) {
 581| 		/*
 582| 		 * If "updatekey" was true for all keys found in the DNSKEY
 583| 		 * response and the previous update of those keys happened
 584| 		 * during the same second (only possible if a key refresh was
 585| 		 * externally triggered), it may happen that all relevant
 586| 		 * update_one_rr() calls will return ISC_R_SUCCESS, but
 587| 		 * diff.tuples will remain empty.  Reset result to
 588| 		 * ISC_R_SUCCESS to prevent a bogus warning from being logged.
 589| 		 */
 590| 		result = ISC_R_SUCCESS;
 591| 	}
 592| 
 593|  failure:
 594| 	if (result != ISC_R_SUCCESS) {
 595| 		dnssec_log(zone, ISC_LOG_ERROR,
 596| 			   "error during managed-keys processing (%s): "
 597| 			   "DNSSEC validation may be at risk",
 598| 			   isc_result_totext(result));
 599| 	}
 600| 	dns_diff_clear(&diff);
 601| 	if (ver != NULL) {
 602| 		dns_db_closeversion(kfetch->db, &ver, commit);
 603| 	}
 604| 
 605|  cleanup:
 606| 	dns_db_detach(&kfetch->db);
 607| 
 608| 	INSIST(zone->irefs > 0);
 609| 	zone->irefs--;
 610| 	kfetch->zone = NULL;
 611| 
 612| 	if (dns_rdataset_isassociated(&kfetch->keydataset)) {
 613| 		dns_rdataset_disassociate(&kfetch->keydataset);
 614| 	}
 615| 	if (dns_rdataset_isassociated(&kfetch->dnskeyset)) {
 616| 		dns_rdataset_disassociate(&kfetch->dnskeyset);
 617| 	}
 618| 	if (dns_rdataset_isassociated(&kfetch->dnskeysigset)) {
 619| 		dns_rdataset_disassociate(&kfetch->dnskeysigset);
 620| 	}
 621| 
 622| 	dns_name_free(keyname, mctx);
 623| 	isc_mem_put(mctx, kfetch, sizeof(dns_keyfetch_t));
 624| 	isc_mem_detach(&mctx);
 625| 
 626| 	if (secroots != NULL) {
 627| 		dns_keytable_detach(&secroots);
 628| 	}
 629| 
 630| 	free_needed = exit_check(zone);
 631| 	UNLOCK_ZONE(zone);
 632| 	if (free_needed) {
 633| 		zone_free(zone);
 634| 	}
 635| 
 636| 	INSIST(ver == NULL);
 637| }
