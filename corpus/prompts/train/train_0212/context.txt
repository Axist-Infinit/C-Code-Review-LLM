   2| bool IsConstantFoldable(
   3|     const Node* n,
   4|     const std::unordered_map<string, std::vector<PartialTensorShape>>*
   5|         shape_map,
   6|     const std::function<bool(const Node*)>& consider,
   7|     int64_t max_constant_size_in_bytes,
   8|     std::unordered_map<const Node*, std::vector<Tensor>>*
   9|         shape_replacement_map) {
  10|   if (n->IsConstant()) {
  11|     return true;
  12|   }
  13|   if (MaybeReplaceShapeOp(n, shape_map, shape_replacement_map)) {
  14|     return true;
  15|   }
  16|   if (n->op_def().is_stateful()) {
  17|     return false;
  18|   }
  19|   if (consider && !consider(n)) {
  20|     return false;
  21|   }
  22|   if (shape_map != nullptr) {
  23|     // We can skip the node if an output is known to be oversized.
  24|     auto shape_it = shape_map->find(n->name());
  25|     if (shape_it != shape_map->end()) {
  26|       for (int64_t i = 0; i < shape_it->second.size(); ++i) {
  27|         const auto& out_shape = shape_it->second[i];
  28|         if (out_shape.IsFullyDefined() &&
  29|             out_shape.num_elements() * DataTypeSize(n->output_type(i)) >
  30|                 max_constant_size_in_bytes) {
  31|           return false;
  32|         }
  33|       }
  34|     }
  35|   }
  36|   if (n->IsControlFlow() || n->IsSend() || n->IsRecv()) {
  37|     return false;
  38|   }
  39|   // TODO(yuanbyu): For now disable these session handle operations.
  40|   if (n->IsGetSessionHandle() || n->IsGetSessionTensor() ||
  41|       n->IsDeleteSessionTensor()) {
  42|     return false;
  43|   }
  44|   if (n->IsSource()) {
  45|     return false;
  46|   }
  47|   if (n->IsSink()) {
  48|     return false;
  49|   }
  50|   if (n->IsFakeParam()) {
  51|     return false;
  52|   }
  53|   // Since constant-folding runs on the CPU, do not attempt to constant-fold
  54|   // operators that have no CPU kernel. Also implies that we will not
  55|   // constant-fold functions.
  56|   // TODO(phawkins): allow constant-folding for functions; functions may
  57|   // be arbitrarily expensive to execute.
  58|   if (!KernelDefAvailable(DeviceType(DEVICE_CPU), n->def())) {
  59|     return false;
  60|   }
  61|   // Do not constant fold nodes which will be allocated by ScopedAllocator.
  62|   // This is because the constant-folding graph will not contain the
  63|   // `_ScopedAllocator` node, and that is necessary to be able to run a node
  64|   // that will use this allocator.
  65|   if (n->attrs().Find(kScopedAllocatorAttrName) != nullptr) {
  66|     VLOG(2) << "Skip node [" << n->DebugString()
  67|             << "] for constant folding due to scoped allocator";
  68|     return false;
  69|   }
  70|   return true;
  71| }
