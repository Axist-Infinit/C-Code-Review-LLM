  18| 	rc = static_config_buf_prepare_for_upload(priv, config_buf, buf_len);

  20| 		dev_err(dev, "Invalid config, cannot upload\n");
  21| 		rc = -EINVAL;
  22| 		goto out;
  23| 	}
  24| 	/* Prevent PHY jabbering during switch reset by inhibiting

  26| 	 * Otherwise, the PHY will see an unterminated Ethernet packet.

  28| 	rc = sja1105_inhibit_tx(priv, port_bitmap, true);

  44| 			continue;
  45| 		}

  47| 		usleep_range(1000, 5000);

  49| 		rc = sja1105_spi_send_long_packed_buf(priv, SPI_WRITE,
  50| 						      regs->config,

  53| 			dev_err(dev, "Failed to upload config, retrying...\n");
  54| 			continue;

  79| 				"invalid, retrying...\n");

  92| 	}
