 468| 				if (src_reg == BPF_REG_0)
 469| 					real_src_reg = BPF_REG_AX;
 470| 
 471| 				branch_target = prog;
 472| 				/* Load old value */
 473| 				emit_ldx(&prog, BPF_SIZE(insn->code),
 474| 					 BPF_REG_0, dst_reg, insn->off);
 475| 				/*
 476| 				 * Perform the (commutative) operation locally,
 477| 				 * put the result in the AUX_REG.
 478| 				 */
 479| 				emit_mov_reg(&prog, is64, AUX_REG, BPF_REG_0);
 480| 				maybe_emit_mod(&prog, AUX_REG, real_src_reg, is64);
 481| 				EMIT2(simple_alu_opcodes[BPF_OP(insn->imm)],
 482| 				      add_2reg(0xC0, AUX_REG, real_src_reg));
 483| 				/* Attempt to swap in new value */
 484| 				err = emit_atomic(&prog, BPF_CMPXCHG,
 485| 						  dst_reg, AUX_REG, insn->off,
 486| 						  BPF_SIZE(insn->code));
 487| 				if (WARN_ON(err))
 488| 					return err;
 489| 				/*
 490| 				 * ZF tells us whether we won the race. If it's
 491| 				 * cleared we need to try again.
 492| 				 */
 493| 				EMIT2(X86_JNE, -(prog - branch_target) - 2);
 494| 				/* Return the pre-modification value */
 495| 				emit_mov_reg(&prog, is64, real_src_reg, BPF_REG_0);
 496| 				/* Restore R0 after clobbering RAX */
 497| 				emit_mov_reg(&prog, true, BPF_REG_0, BPF_REG_AX);
 498| 				break;
 499| 
 500| 			}
 501| 
 502| 			err = emit_atomic(&prog, insn->imm, dst_reg, src_reg,
 503| 						  insn->off, BPF_SIZE(insn->code));
 504| 			if (err)
 505| 				return err;
 506| 			break;
 507| 
 508| 			/* call */
 509| 		case BPF_JMP | BPF_CALL:
 510| 			func = (u8 *) __bpf_call_base + imm32;
 511| 			if (tail_call_reachable) {
 512| 				EMIT3_off32(0x48, 0x8B, 0x85,
 513| 					    -(bpf_prog->aux->stack_depth + 8));
 514| 				if (!imm32 || emit_call(&prog, func, image + addrs[i - 1] + 7))
 515| 					return -EINVAL;
 516| 			} else {
 517| 				if (!imm32 || emit_call(&prog, func, image + addrs[i - 1]))
 518| 					return -EINVAL;
 519| 			}
 520| 			break;
 521| 
 522| 		case BPF_JMP | BPF_TAIL_CALL:
 523| 			if (imm32)
 524| 				emit_bpf_tail_call_direct(&bpf_prog->aux->poke_tab[imm32 - 1],
 525| 							  &prog, addrs[i], image,
 526| 							  callee_regs_used,
 527| 							  bpf_prog->aux->stack_depth);
 528| 			else
 529| 				emit_bpf_tail_call_indirect(&prog,
 530| 							    callee_regs_used,
 531| 							    bpf_prog->aux->stack_depth);
 532| 			break;
 533| 
 534| 			/* cond jump */
 535| 		case BPF_JMP | BPF_JEQ | BPF_X:
 536| 		case BPF_JMP | BPF_JNE | BPF_X:
 537| 		case BPF_JMP | BPF_JGT | BPF_X:
 538| 		case BPF_JMP | BPF_JLT | BPF_X:
 539| 		case BPF_JMP | BPF_JGE | BPF_X:
 540| 		case BPF_JMP | BPF_JLE | BPF_X:
 541| 		case BPF_JMP | BPF_JSGT | BPF_X:
 542| 		case BPF_JMP | BPF_JSLT | BPF_X:
 543| 		case BPF_JMP | BPF_JSGE | BPF_X:
 544| 		case BPF_JMP | BPF_JSLE | BPF_X:
 545| 		case BPF_JMP32 | BPF_JEQ | BPF_X:
 546| 		case BPF_JMP32 | BPF_JNE | BPF_X:
 547| 		case BPF_JMP32 | BPF_JGT | BPF_X:
 548| 		case BPF_JMP32 | BPF_JLT | BPF_X:
 549| 		case BPF_JMP32 | BPF_JGE | BPF_X:
 550| 		case BPF_JMP32 | BPF_JLE | BPF_X:
 551| 		case BPF_JMP32 | BPF_JSGT | BPF_X:
 552| 		case BPF_JMP32 | BPF_JSLT | BPF_X:
 553| 		case BPF_JMP32 | BPF_JSGE | BPF_X:
 554| 		case BPF_JMP32 | BPF_JSLE | BPF_X:
 555| 			/* cmp dst_reg, src_reg */
 556| 			maybe_emit_mod(&prog, dst_reg, src_reg,
 557| 				       BPF_CLASS(insn->code) == BPF_JMP);
 558| 			EMIT2(0x39, add_2reg(0xC0, dst_reg, src_reg));
 559| 			goto emit_cond_jmp;
 560| 
 561| 		case BPF_JMP | BPF_JSET | BPF_X:
 562| 		case BPF_JMP32 | BPF_JSET | BPF_X:
 563| 			/* test dst_reg, src_reg */
 564| 			maybe_emit_mod(&prog, dst_reg, src_reg,
 565| 				       BPF_CLASS(insn->code) == BPF_JMP);
 566| 			EMIT2(0x85, add_2reg(0xC0, dst_reg, src_reg));
 567| 			goto emit_cond_jmp;
 568| 
 569| 		case BPF_JMP | BPF_JSET | BPF_K:
 570| 		case BPF_JMP32 | BPF_JSET | BPF_K:
 571| 			/* test dst_reg, imm32 */
 572| 			if (BPF_CLASS(insn->code) == BPF_JMP)
 573| 				EMIT1(add_1mod(0x48, dst_reg));
 574| 			else if (is_ereg(dst_reg))
 575| 				EMIT1(add_1mod(0x40, dst_reg));
 576| 			EMIT2_off32(0xF7, add_1reg(0xC0, dst_reg), imm32);
 577| 			goto emit_cond_jmp;
 578| 
 579| 		case BPF_JMP | BPF_JEQ | BPF_K:
 580| 		case BPF_JMP | BPF_JNE | BPF_K:
 581| 		case BPF_JMP | BPF_JGT | BPF_K:
 582| 		case BPF_JMP | BPF_JLT | BPF_K:
 583| 		case BPF_JMP | BPF_JGE | BPF_K:
 584| 		case BPF_JMP | BPF_JLE | BPF_K:
 585| 		case BPF_JMP | BPF_JSGT | BPF_K:
 586| 		case BPF_JMP | BPF_JSLT | BPF_K:
 587| 		case BPF_JMP | BPF_JSGE | BPF_K:
 588| 		case BPF_JMP | BPF_JSLE | BPF_K:
 589| 		case BPF_JMP32 | BPF_JEQ | BPF_K:
 590| 		case BPF_JMP32 | BPF_JNE | BPF_K:
 591| 		case BPF_JMP32 | BPF_JGT | BPF_K:
 592| 		case BPF_JMP32 | BPF_JLT | BPF_K:
 593| 		case BPF_JMP32 | BPF_JGE | BPF_K:
 594| 		case BPF_JMP32 | BPF_JLE | BPF_K:
 595| 		case BPF_JMP32 | BPF_JSGT | BPF_K:
 596| 		case BPF_JMP32 | BPF_JSLT | BPF_K:
 597| 		case BPF_JMP32 | BPF_JSGE | BPF_K:
 598| 		case BPF_JMP32 | BPF_JSLE | BPF_K:
 599| 			/* test dst_reg, dst_reg to save one extra byte */
 600| 			if (imm32 == 0) {
 601| 				maybe_emit_mod(&prog, dst_reg, dst_reg,
 602| 					       BPF_CLASS(insn->code) == BPF_JMP);
 603| 				EMIT2(0x85, add_2reg(0xC0, dst_reg, dst_reg));
 604| 				goto emit_cond_jmp;
 605| 			}
 606| 
 607| 			/* cmp dst_reg, imm8/32 */
 608| 			if (BPF_CLASS(insn->code) == BPF_JMP)
 609| 				EMIT1(add_1mod(0x48, dst_reg));
 610| 			else if (is_ereg(dst_reg))
 611| 				EMIT1(add_1mod(0x40, dst_reg));
 612| 
 613| 			if (is_imm8(imm32))
 614| 				EMIT3(0x83, add_1reg(0xF8, dst_reg), imm32);
 615| 			else
 616| 				EMIT2_off32(0x81, add_1reg(0xF8, dst_reg), imm32);
 617| 
 618| emit_cond_jmp:		/* Convert BPF opcode to x86 */
 619| 			switch (BPF_OP(insn->code)) {
 620| 			case BPF_JEQ:
 621| 				jmp_cond = X86_JE;
 622| 				break;
 623| 			case BPF_JSET:
 624| 			case BPF_JNE:
 625| 				jmp_cond = X86_JNE;
 626| 				break;
 627| 			case BPF_JGT:
 628| 				/* GT is unsigned '>', JA in x86 */
 629| 				jmp_cond = X86_JA;
 630| 				break;
 631| 			case BPF_JLT:
 632| 				/* LT is unsigned '<', JB in x86 */
 633| 				jmp_cond = X86_JB;
 634| 				break;
 635| 			case BPF_JGE:
 636| 				/* GE is unsigned '>=', JAE in x86 */
 637| 				jmp_cond = X86_JAE;
 638| 				break;
 639| 			case BPF_JLE:
 640| 				/* LE is unsigned '<=', JBE in x86 */
 641| 				jmp_cond = X86_JBE;
 642| 				break;
 643| 			case BPF_JSGT:
 644| 				/* Signed '>', GT in x86 */
 645| 				jmp_cond = X86_JG;
 646| 				break;
 647| 			case BPF_JSLT:
 648| 				/* Signed '<', LT in x86 */
 649| 				jmp_cond = X86_JL;
 650| 				break;
 651| 			case BPF_JSGE:
 652| 				/* Signed '>=', GE in x86 */
 653| 				jmp_cond = X86_JGE;
 654| 				break;
 655| 			case BPF_JSLE:
 656| 				/* Signed '<=', LE in x86 */
 657| 				jmp_cond = X86_JLE;
 658| 				break;
 659| 			default: /* to silence GCC warning */
 660| 				return -EFAULT;
 661| 			}
 662| 			jmp_offset = addrs[i + insn->off] - addrs[i];
 663| 			if (is_imm8(jmp_offset)) {
 664| 				if (jmp_padding) {
 665| 					/* To keep the jmp_offset valid, the extra bytes are
 666| 					 * padded before the jump insn, so we substract the
 667| 					 * 2 bytes of jmp_cond insn from INSN_SZ_DIFF.
 668| 					 *
 669| 					 * If the previous pass already emits an imm8
 670| 					 * jmp_cond, then this BPF insn won't shrink, so
 671| 					 * "nops" is 0.
 672| 					 *
 673| 					 * On the other hand, if the previous pass emits an
 674| 					 * imm32 jmp_cond, the extra 4 bytes(*) is padded to
 675| 					 * keep the image from shrinking further.
 676| 					 *
 677| 					 * (*) imm32 jmp_cond is 6 bytes, and imm8 jmp_cond
 678| 					 *     is 2 bytes, so the size difference is 4 bytes.
 679| 					 */
 680| 					nops = INSN_SZ_DIFF - 2;
 681| 					if (nops != 0 && nops != 4) {
 682| 						pr_err("unexpected jmp_cond padding: %d bytes\n",
 683| 						       nops);
 684| 						return -EFAULT;
 685| 					}
 686| 					cnt += emit_nops(&prog, nops);
 687| 				}
 688| 				EMIT2(jmp_cond, jmp_offset);
 689| 			} else if (is_simm32(jmp_offset)) {
 690| 				EMIT2_off32(0x0F, jmp_cond + 0x10, jmp_offset);
 691| 			} else {
 692| 				pr_err("cond_jmp gen bug %llx\n", jmp_offset);
 693| 				return -EFAULT;
 694| 			}
 695| 
 696| 			break;
 697| 
 698| 		case BPF_JMP | BPF_JA:
 699| 			if (insn->off == -1)
 700| 				/* -1 jmp instructions will always jump
 701| 				 * backwards two bytes. Explicitly handling
 702| 				 * this case avoids wasting too many passes
 703| 				 * when there are long sequences of replaced
 704| 				 * dead code.
 705| 				 */
 706| 				jmp_offset = -2;
 707| 			else
 708| 				jmp_offset = addrs[i + insn->off] - addrs[i];
 709| 
 710| 			if (!jmp_offset) {
 711| 				/*
 712| 				 * If jmp_padding is enabled, the extra nops will
 713| 				 * be inserted. Otherwise, optimize out nop jumps.
 714| 				 */
 715| 				if (jmp_padding) {
 716| 					/* There are 3 possible conditions.
 717| 					 * (1) This BPF_JA is already optimized out in
 718| 					 *     the previous run, so there is no need
 719| 					 *     to pad any extra byte (0 byte).
 720| 					 * (2) The previous pass emits an imm8 jmp,
 721| 					 *     so we pad 2 bytes to match the previous
 722| 					 *     insn size.
 723| 					 * (3) Similarly, the previous pass emits an
 724| 					 *     imm32 jmp, and 5 bytes is padded.
 725| 					 */
 726| 					nops = INSN_SZ_DIFF;
 727| 					if (nops != 0 && nops != 2 && nops != 5) {
 728| 						pr_err("unexpected nop jump padding: %d bytes\n",
 729| 						       nops);
 730| 						return -EFAULT;
 731| 					}
 732| 					cnt += emit_nops(&prog, nops);
 733| 				}
 734| 				break;
 735| 			}
 736| emit_jmp:
 737| 			if (is_imm8(jmp_offset)) {
 738| 				if (jmp_padding) {
 739| 					/* To avoid breaking jmp_offset, the extra bytes
 740| 					 * are padded before the actual jmp insn, so
 741| 					 * 2 bytes is substracted from INSN_SZ_DIFF.
 742| 					 *
 743| 					 * If the previous pass already emits an imm8
 744| 					 * jmp, there is nothing to pad (0 byte).
 745| 					 *
 746| 					 * If it emits an imm32 jmp (5 bytes) previously
 747| 					 * and now an imm8 jmp (2 bytes), then we pad
 748| 					 * (5 - 2 = 3) bytes to stop the image from
 749| 					 * shrinking further.
 750| 					 */
 751| 					nops = INSN_SZ_DIFF - 2;
 752| 					if (nops != 0 && nops != 3) {
 753| 						pr_err("unexpected jump padding: %d bytes\n",
 754| 						       nops);
 755| 						return -EFAULT;
 756| 					}
 757| 					cnt += emit_nops(&prog, INSN_SZ_DIFF - 2);
 758| 				}
 759| 				EMIT2(0xEB, jmp_offset);
 760| 			} else if (is_simm32(jmp_offset)) {
 761| 				EMIT1_off32(0xE9, jmp_offset);
 762| 			} else {
 763| 				pr_err("jmp gen bug %llx\n", jmp_offset);
 764| 				return -EFAULT;
 765| 			}
 766| 			break;
 767| 
 768| 		case BPF_JMP | BPF_EXIT:
 769| 			if (seen_exit) {
 770| 				jmp_offset = ctx->cleanup_addr - addrs[i];
 771| 				goto emit_jmp;
 772| 			}
 773| 			seen_exit = true;
 774| 			/* Update cleanup_addr */
 775| 			ctx->cleanup_addr = proglen;
 776| 			pop_callee_regs(&prog, callee_regs_used);
 777| 			EMIT1(0xC9);         /* leave */
 778| 			EMIT1(0xC3);         /* ret */
 779| 			break;
 780| 
 781| 		default:
 782| 			/*
 783| 			 * By design x86-64 JIT should support all BPF instructions.
 784| 			 * This error will be seen if new instruction was added
 785| 			 * to the interpreter, but not to the JIT, or if there is
 786| 			 * junk in bpf_prog.
 787| 			 */
 788| 			pr_err("bpf_jit: unknown opcode %02x\n", insn->code);
 789| 			return -EINVAL;
 790| 		}
 791| 
 792| 		ilen = prog - temp;
 793| 		if (ilen > BPF_MAX_INSN_SIZE) {
 794| 			pr_err("bpf_jit: fatal insn size error\n");
 795| 			return -EFAULT;
 796| 		}
 797| 
 798| 		if (image) {
 799| 			if (unlikely(proglen + ilen > oldproglen)) {
 800| 				pr_err("bpf_jit: fatal error\n");
 801| 				return -EFAULT;
 802| 			}
 803| 			memcpy(image + proglen, temp, ilen);
 804| 		}
 805| 		proglen += ilen;
 806| 		addrs[i] = proglen;
 807| 		prog = temp;
 808| 	}
 809| 
 810| 	if (image && excnt != bpf_prog->aux->num_exentries) {
 811| 		pr_err("extable is not populated\n");
 812| 		return -EFAULT;
 813| 	}
 814| 	return proglen;
 815| }
