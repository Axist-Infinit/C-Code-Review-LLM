   2| QPDFObjectHandle::parseInternal(PointerHolder<InputSource> input,
   3|                                 std::string const& object_description,
   4|                                 QPDFTokenizer& tokenizer, bool& empty,
   5|                                 StringDecrypter* decrypter, QPDF* context,
   6|                                 bool content_stream)
   7| {
   8|     // This method must take care not to resolve any objects. Don't
   9|     // check the type of any object without first ensuring that it is
  10|     // a direct object. Otherwise, doing so may have the side effect
  11|     // of reading the object and changing the file pointer.
  12| 
  13|     empty = false;
  14| 
  15|     QPDFObjectHandle object;
  16| 
  17|     std::vector<std::vector<QPDFObjectHandle> > olist_stack;
  18|     olist_stack.push_back(std::vector<QPDFObjectHandle>());
  19|     enum state_e { st_top, st_start, st_stop, st_eof, st_dictionary, st_array };
  20|     std::vector<state_e> state_stack;
  21|     state_stack.push_back(st_top);
  22|     std::vector<qpdf_offset_t> offset_stack;
  23|     offset_stack.push_back(input->tell());
  24|     bool done = false;
  25|     while (! done)
  26|     {
  27|         std::vector<QPDFObjectHandle>& olist = olist_stack.back();
  28|         state_e state = state_stack.back();
  29|         qpdf_offset_t offset = offset_stack.back();
  30| 
  31| 	object = QPDFObjectHandle();
  32| 
  33| 	QPDFTokenizer::Token token =
  34|             tokenizer.readToken(input, object_description);
  35| 
  36| 	switch (token.getType())
  37| 	{
  38|           case QPDFTokenizer::tt_eof:
  39|             if (content_stream)
  40|             {
  41|                 state = st_eof;
  42|             }
  43|             else
  44|             {
  45|                 // When not in content stream mode, EOF is tt_bad and
  46|                 // throws an exception before we get here.
  47|                 throw std::logic_error(
  48|                     "EOF received while not in content stream mode");
  49|             }
  50|             break;
  51| 
  52| 	  case QPDFTokenizer::tt_brace_open:
  53| 	  case QPDFTokenizer::tt_brace_close:
  54| 	    QTC::TC("qpdf", "QPDFObjectHandle bad brace");
  55|             warn(context,
  56|                  QPDFExc(qpdf_e_damaged_pdf, input->getName(),
  57|                          object_description,
  58|                          input->getLastOffset(),
  59|                          "treating unexpected brace token as null"));
  60|             object = newNull();
  61| 	    break;
  62| 
  63| 	  case QPDFTokenizer::tt_array_close:
  64| 	    if (state == st_array)
  65| 	    {
  66|                 state = st_stop;
  67| 	    }
  68| 	    else
  69| 	    {
  70| 		QTC::TC("qpdf", "QPDFObjectHandle bad array close");
  71|                 warn(context,
  72|                      QPDFExc(qpdf_e_damaged_pdf, input->getName(),
  73|                              object_description,
  74|                              input->getLastOffset(),
  75|                              "treating unexpected array close token as null"));
  76|                 object = newNull();
  77| 	    }
  78| 	    break;
  79| 
  80| 	  case QPDFTokenizer::tt_dict_close:
  81| 	    if (state == st_dictionary)
  82| 	    {
  83|                 state = st_stop;
  84| 	    }
  85| 	    else
  86| 	    {
  87| 		QTC::TC("qpdf", "QPDFObjectHandle bad dictionary close");
  88|                 warn(context,
  89|                      QPDFExc(qpdf_e_damaged_pdf, input->getName(),
  90|                              object_description,
  91|                              input->getLastOffset(),
  92|                              "unexpected dictionary close token"));
  93|                 object = newNull();
  94| 	    }
  95| 	    break;
  96| 
  97| 	  case QPDFTokenizer::tt_array_open:
  98| 	  case QPDFTokenizer::tt_dict_open:
  99|             olist_stack.push_back(std::vector<QPDFObjectHandle>());
 100|             state = st_start;
 101|             offset_stack.push_back(input->tell());
 102|             state_stack.push_back(
 103|                 (token.getType() == QPDFTokenizer::tt_array_open) ?
 104|                 st_array : st_dictionary);
 105| 	    break;
 106| 
 107| 	  case QPDFTokenizer::tt_bool:
 108| 	    object = newBool((token.getValue() == "true"));
 109| 	    break;
 110| 
 111| 	  case QPDFTokenizer::tt_null:
 112| 	    object = newNull();
 113| 	    break;
 114| 
 115| 	  case QPDFTokenizer::tt_integer:
 116| 	    object = newInteger(QUtil::string_to_ll(token.getValue().c_str()));
 117| 	    break;
 118| 
 119| 	  case QPDFTokenizer::tt_real:
 120| 	    object = newReal(token.getValue());
 121| 	    break;
 122| 
 123| 	  case QPDFTokenizer::tt_name:
 124| 	    object = newName(token.getValue());
 125| 	    break;
 126| 
 127| 	  case QPDFTokenizer::tt_word:
 128| 	    {
 129| 		std::string const& value = token.getValue();
 130|                 if (content_stream)
 131|                 {
 132|                     object = QPDFObjectHandle::newOperator(value);
 133|                 }
 134| 		else if ((value == "R") && (state != st_top) &&
 135|                          (olist.size() >= 2) &&
 136|                          (! olist.at(olist.size() - 1).isIndirect()) &&
 137|                          (olist.at(olist.size() - 1).isInteger()) &&
 138|                          (! olist.at(olist.size() - 2).isIndirect()) &&
 139|                          (olist.at(olist.size() - 2).isInteger()))
 140| 		{
 141|                     if (context == 0)
 142|                     {
 143|                         QTC::TC("qpdf", "QPDFObjectHandle indirect without context");
 144|                         throw std::logic_error(
 145|                             "QPDFObjectHandle::parse called without context"
 146|                             " on an object with indirect references");
 147|                     }
 148| 		    // Try to resolve indirect objects
 149| 		    object = newIndirect(
 150| 			context,
 151| 			olist.at(olist.size() - 2).getIntValue(),
 152| 			olist.at(olist.size() - 1).getIntValue());
 153| 		    olist.pop_back();
 154| 		    olist.pop_back();
 155| 		}
 156| 		else if ((value == "endobj") && (state == st_top))
 157| 		{
 158| 		    // We just saw endobj without having read
 159| 		    // anything.  Treat this as a null and do not move
 160| 		    // the input source's offset.
 161| 		    object = newNull();
 162| 		    input->seek(input->getLastOffset(), SEEK_SET);
 163|                     empty = true;
 164| 		}
 165| 		else
 166| 		{
 167|                     QTC::TC("qpdf", "QPDFObjectHandle treat word as string");
 168|                     warn(context,
 169|                          QPDFExc(qpdf_e_damaged_pdf, input->getName(),
 170|                                  object_description,
 171|                                  input->getLastOffset(),
 172|                                  "unknown token while reading object;"
 173|                                  " treating as string"));
 174|                     object = newString(value);
 175| 		}
 176| 	    }
 177| 	    break;
 178| 
 179| 	  case QPDFTokenizer::tt_string:
 180| 	    {
 181| 		std::string val = token.getValue();
 182|                 if (decrypter)
 183|                 {
 184|                     decrypter->decryptString(val);
 185|                 }
 186| 		object = QPDFObjectHandle::newString(val);
 187| 	    }
 188| 
 189| 	    break;
 190| 
 191| 	  default:
 192|             warn(context,
 193|                  QPDFExc(qpdf_e_damaged_pdf, input->getName(),
 194|                          object_description,
 195|                          input->getLastOffset(),
 196|                          "treating unknown token type as null while "
 197|                          "reading object"));
 198|             object = newNull();
 199| 	    break;
 200| 	}
 201| 
 202|         if ((! object.isInitialized()) &&
 203|             (! ((state == st_start) ||
 204|                 (state == st_stop) ||
 205|                 (state == st_eof))))
 206|         {
 207|             throw std::logic_error(
 208|                 "QPDFObjectHandle::parseInternal: "
 209|                 "unexpected uninitialized object");
 210|             object = newNull();
 211|         }
 212| 
 213|         switch (state)
 214|         {
 215|           case st_eof:
 216|             if (state_stack.size() > 1)
 217|             {
 218|                 warn(context,
 219|                      QPDFExc(qpdf_e_damaged_pdf, input->getName(),
 220|                              object_description,
 221|                              input->getLastOffset(),
 222|                              "parse error while reading object"));
 223|             }
 224|             done = true;
 225|             // Leave object uninitialized to indicate EOF
 226|             break;
 227| 
 228|           case st_dictionary:
 229|           case st_array:
 230|             olist.push_back(object);
 231|             break;
 232| 
 233|           case st_top:
 234|             done = true;
 235|             break;
 236| 
 237|           case st_start:
 238|             break;
 239| 
 240|           case st_stop:
 241|             if ((state_stack.size() < 2) || (olist_stack.size() < 2))
 242|             {
 243|                 throw std::logic_error(
 244|                     "QPDFObjectHandle::parseInternal: st_stop encountered"
 245|                     " with insufficient elements in stack");
 246|             }
 247|             state_e old_state = state_stack.back();
 248|             state_stack.pop_back();
 249|             if (old_state == st_array)
 250|             {
 251|                 object = newArray(olist);
 252|             }
 253|             else if (old_state == st_dictionary)
 254|             {
 255|                 // Convert list to map. Alternating elements are keys.
 256|                 // Attempt to recover more or less gracefully from
 257|                 // invalid dictionaries.
 258|                 std::set<std::string> names;
 259|                 for (std::vector<QPDFObjectHandle>::iterator iter =
 260|                          olist.begin();
 261|                      iter != olist.end(); ++iter)
 262|                 {
 263|                     if ((! (*iter).isIndirect()) && (*iter).isName())
 264|                     {
 265|                         names.insert((*iter).getName());
 266|                     }
 267|                 }
 268| 
 269|                 std::map<std::string, QPDFObjectHandle> dict;
 270|                 int next_fake_key = 1;
 271|                 for (unsigned int i = 0; i < olist.size(); ++i)
 272|                 {
 273|                     QPDFObjectHandle key_obj = olist.at(i);
 274|                     QPDFObjectHandle val;
 275|                     if (key_obj.isIndirect() || (! key_obj.isName()))
 276|                     {
 277|                         bool found_fake = false;
 278|                         std::string candidate;
 279|                         while (! found_fake)
 280|                         {
 281|                             candidate =
 282|                                 "/QPDFFake" +
 283|                                 QUtil::int_to_string(next_fake_key++);
 284|                             found_fake = (names.count(candidate) == 0);
 285|                             QTC::TC("qpdf", "QPDFObjectHandle found fake",
 286|                                     (found_fake ? 0 : 1));
 287|                         }
 288|                         warn(context,
 289|                              QPDFExc(
 290|                                  qpdf_e_damaged_pdf,
 291|                                  input->getName(), object_description, offset,
 292|                                  "expected dictionary key but found"
 293|                                  " non-name object; inserting key " +
 294|                                  candidate));
 295|                         val = key_obj;
 296|                         key_obj = newName(candidate);
 297|                     }
 298|                     else if (i + 1 >= olist.size())
 299|                     {
 300|                         QTC::TC("qpdf", "QPDFObjectHandle no val for last key");
 301|                         warn(context,
 302|                              QPDFExc(
 303|                                  qpdf_e_damaged_pdf,
 304|                                  input->getName(), object_description, offset,
 305|                                  "dictionary ended prematurely; "
 306|                                  "using null as value for last key"));
 307|                         val = newNull();
 308|                     }
 309|                     else
 310|                     {
 311|                         val = olist.at(++i);
 312|                     }
 313|                     dict[key_obj.getName()] = val;
 314|                 }
 315|                 object = newDictionary(dict);
 316|             }
 317|             olist_stack.pop_back();
 318|             offset_stack.pop_back();
 319|             if (state_stack.back() == st_top)
 320|             {
 321|                 done = true;
 322|             }
 323|             else
 324|             {
 325|                 olist_stack.back().push_back(object);
 326|             }
 327|         }
 328|     }
 329| 
 330|     return object;
 331| }
