   2|   void Compute(OpKernelContext* context) override {
   3|     // Here's the basic idea:
   4|     // Batch and depth dimension are independent from row and col dimension. And
   5|     // because FractionalAvgPool currently only support pooling along row and
   6|     // col, we can basically think of this 4D tensor backpropagation as
   7|     // operation of a series of 2D planes.
   8|     //
   9|     // For each element of a 'slice' (2D plane) of output_backprop, we need to
  10|     // figure out its contributors when doing FractionalAvgPool operation. This
  11|     // can be done based on row_pooling_sequence, col_pooling_seq and
  12|     // overlapping.
  13|     // Once we figure out the original contributors, we just need to evenly
  14|     // divide the value of this element among these contributors.
  15|     //
  16|     // Internally, we divide the out_backprop tensor and store it in a temporary
  17|     // tensor of double type. And cast it to the corresponding type.
  18|     typedef Eigen::Map<const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>>
  19|         ConstEigenMatrixMap;
  20|     typedef Eigen::Map<Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic>>
  21|         EigenDoubleMatrixMap;
  22| 
  23|     // Grab the inputs.
  24|     const Tensor& orig_input_tensor_shape = context->input(0);
  25|     OP_REQUIRES(context,
  26|                 orig_input_tensor_shape.dims() == 1 &&
  27|                     orig_input_tensor_shape.NumElements() == 4,
  28|                 errors::InvalidArgument("original input tensor shape must be"
  29|                                         "1-dimensional and 4 elements"));
  30|     const Tensor& out_backprop = context->input(1);
  31|     const Tensor& row_seq_tensor = context->input(2);
  32|     const Tensor& col_seq_tensor = context->input(3);
  33| 
  34|     const int64 out_batch = out_backprop.dim_size(0);
  35|     const int64 out_rows = out_backprop.dim_size(1);
  36|     const int64 out_cols = out_backprop.dim_size(2);
  37|     const int64 out_depth = out_backprop.dim_size(3);
  38| 
  39|     OP_REQUIRES(context, row_seq_tensor.NumElements() > out_rows,
  40|                 errors::InvalidArgument("Given out_backprop shape ",
  41|                                         out_backprop.shape().DebugString(),
  42|                                         ", row_seq_tensor must have at least ",
  43|                                         out_rows + 1, " elements, but got ",
  44|                                         row_seq_tensor.NumElements()));
  45|     OP_REQUIRES(context, col_seq_tensor.NumElements() > out_cols,
  46|                 errors::InvalidArgument("Given out_backprop shape ",
  47|                                         out_backprop.shape().DebugString(),
  48|                                         ", col_seq_tensor must have at least ",
  49|                                         out_cols + 1, " elements, but got ",
  50|                                         col_seq_tensor.NumElements()));
  51| 
  52|     auto row_seq_tensor_flat = row_seq_tensor.flat<int64>();
  53|     auto col_seq_tensor_flat = col_seq_tensor.flat<int64>();
  54|     auto orig_input_tensor_shape_flat = orig_input_tensor_shape.flat<int64>();
  55| 
  56|     const int64 in_batch = orig_input_tensor_shape_flat(0);
  57|     const int64 in_rows = orig_input_tensor_shape_flat(1);
  58|     const int64 in_cols = orig_input_tensor_shape_flat(2);
  59|     const int64 in_depth = orig_input_tensor_shape_flat(3);
  60| 
  61|     constexpr int tensor_in_and_out_dims = 4;
  62|     // Transform orig_input_tensor_shape into TensorShape
  63|     TensorShape in_shape;
  64|     for (auto i = 0; i < tensor_in_and_out_dims; ++i) {
  65|       in_shape.AddDim(orig_input_tensor_shape_flat(i));
  66|     }
  67| 
  68|     // Create intermediate in_backprop.
  69|     Tensor in_backprop_tensor_temp;
  70|     OP_REQUIRES_OK(context, context->forward_input_or_allocate_temp(
  71|                                 {0}, DataTypeToEnum<double>::v(), in_shape,
  72|                                 &in_backprop_tensor_temp));
  73|     in_backprop_tensor_temp.flat<double>().setZero();
  74|     // Transform 4D tensor to 2D matrix.
  75|     EigenDoubleMatrixMap in_backprop_tensor_temp_mat(
  76|         in_backprop_tensor_temp.flat<double>().data(), in_depth,
  77|         in_cols * in_rows * in_batch);
  78|     ConstEigenMatrixMap out_backprop_mat(out_backprop.flat<T>().data(),
  79|                                          out_depth,
  80|                                          out_cols * out_rows * out_batch);
  81|     // Loop through each element of out_backprop and evenly distribute the
  82|     // element to the corresponding pooling cell.
  83|     const int64 in_max_row_index = in_rows - 1;
  84|     const int64 in_max_col_index = in_cols - 1;
  85|     for (int64 b = 0; b < out_batch; ++b) {
  86|       for (int64 r = 0; r < out_rows; ++r) {
  87|         const int64 in_row_start = row_seq_tensor_flat(r);
  88|         int64 in_row_end = overlapping_ ? row_seq_tensor_flat(r + 1)
  89|                                         : row_seq_tensor_flat(r + 1) - 1;
  90|         in_row_end = std::min(in_row_end, in_max_row_index);
  91|         for (int64 c = 0; c < out_cols; ++c) {
  92|           const int64 in_col_start = col_seq_tensor_flat(c);
  93|           int64 in_col_end = overlapping_ ? col_seq_tensor_flat(c + 1)
  94|                                           : col_seq_tensor_flat(c + 1) - 1;
  95|           in_col_end = std::min(in_col_end, in_max_col_index);
  96| 
  97|           const int64 num_elements_in_pooling_cell =
  98|               (in_row_end - in_row_start + 1) * (in_col_end - in_col_start + 1);
  99|           const int64 out_index = (b * out_rows + r) * out_cols + c;
 100|           // Now we can evenly distribute out_backprop(b, h, w, *) to
 101|           // in_backprop(b, hs:he, ws:we, *).
 102|           for (int64 in_r = in_row_start; in_r <= in_row_end; ++in_r) {
 103|             for (int64 in_c = in_col_start; in_c <= in_col_end; ++in_c) {
 104|               const int64 in_index = (b * in_rows + in_r) * in_cols + in_c;
 105|               // Walk through each channel (depth).
 106|               for (int64 d = 0; d < out_depth; ++d) {
 107|                 const double out_backprop_element = static_cast<double>(
 108|                     out_backprop_mat.coeffRef(d, out_index));
 109|                 double& in_backprop_ref =
 110|                     in_backprop_tensor_temp_mat.coeffRef(d, in_index);
 111|                 in_backprop_ref +=
 112|                     out_backprop_element / num_elements_in_pooling_cell;
 113|               }
 114|             }
 115|           }
 116|         }
 117|       }
 118|     }
 119| 
 120|     // Depending on the type, cast double to type T.
 121|     Tensor* in_backprop_tensor = nullptr;
 122|     OP_REQUIRES_OK(context, context->forward_input_or_allocate_output(
 123|                                 {0}, 0, in_shape, &in_backprop_tensor));
 124|     auto in_backprop_tensor_flat = in_backprop_tensor->flat<T>();
 125|     auto in_backprop_tensor_temp_flat = in_backprop_tensor_temp.flat<double>();
 126|     for (int64 i = 0; i < in_backprop_tensor_flat.size(); ++i) {
 127|       in_backprop_tensor_flat(i) =
 128|           static_cast<T>(in_backprop_tensor_temp_flat(i));
 129|     }
 130|   }
