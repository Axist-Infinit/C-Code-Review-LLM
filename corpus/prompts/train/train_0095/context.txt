   2| static bool split_region(struct uc_struct *uc, MemoryRegion *mr,
   3|                          uint64_t address, size_t size, bool do_delete)
   4| {
   5|     uint8_t *backup;
   6|     uint32_t perms;
   7|     uint64_t begin, end, chunk_end;
   8|     size_t l_size, m_size, r_size;
   9|     RAMBlock *block = NULL;
  10|     bool prealloc = false;
  11| 
  12|     chunk_end = address + size;
  13| 
  14|     // if this region belongs to area [address, address+size],
  15|     // then there is no work to do.
  16|     if (address <= mr->addr && chunk_end >= mr->end) {
  17|         return true;
  18|     }
  19| 
  20|     if (size == 0) {
  21|         // trivial case
  22|         return true;
  23|     }
  24| 
  25|     if (address >= mr->end || chunk_end <= mr->addr) {
  26|         // impossible case
  27|         return false;
  28|     }
  29| 
  30|     QLIST_FOREACH(block, &uc->ram_list.blocks, next)
  31|     {
  32|         // block->offset is the offset within ram_addr_t, not GPA
  33|         if (block->mr->addr <= mr->addr &&
  34|             block->used_length >= (mr->end - mr->addr)) {
  35|             break;
  36|         }
  37|     }
  38| 
  39|     if (block == NULL) {
  40|         return false;
  41|     }
  42| 
  43|     // RAM_PREALLOC is not defined outside exec.c and I didn't feel like
  44|     // moving it
  45|     prealloc = !!(block->flags & 1);
  46| 
  47|     if (block->flags & 1) {
  48|         backup = block->host;
  49|     } else {
  50|         backup = copy_region(uc, mr);
  51|         if (backup == NULL) {
  52|             return false;
  53|         }
  54|     }
  55| 
  56|     // save the essential information required for the split before mr gets
  57|     // deleted
  58|     perms = mr->perms;
  59|     begin = mr->addr;
  60|     end = mr->end;
  61| 
  62|     // unmap this region first, then do split it later
  63|     if (uc_mem_unmap(uc, mr->addr, (size_t)int128_get64(mr->size)) !=
  64|         UC_ERR_OK) {
  65|         goto error;
  66|     }
  67| 
  68|     /* overlapping cases
  69|      *               |------mr------|
  70|      * case 1    |---size--|
  71|      * case 2           |--size--|
  72|      * case 3                  |---size--|
  73|      */
  74| 
  75|     // adjust some things
  76|     if (address < begin) {
  77|         address = begin;
  78|     }
  79|     if (chunk_end > end) {
  80|         chunk_end = end;
  81|     }
  82| 
  83|     // compute sub region sizes
  84|     l_size = (size_t)(address - begin);
  85|     r_size = (size_t)(end - chunk_end);
  86|     m_size = (size_t)(chunk_end - address);
  87| 
  88|     // If there are error in any of the below operations, things are too far
  89|     // gone at that point to recover. Could try to remap orignal region, but
  90|     // these smaller allocation just failed so no guarantee that we can recover
  91|     // the original allocation at this point
  92|     if (l_size > 0) {
  93|         if (!prealloc) {
  94|             if (uc_mem_map(uc, begin, l_size, perms) != UC_ERR_OK) {
  95|                 goto error;
  96|             }
  97|             if (uc_mem_write(uc, begin, backup, l_size) != UC_ERR_OK) {
  98|                 goto error;
  99|             }
 100|         } else {
 101|             if (uc_mem_map_ptr(uc, begin, l_size, perms, backup) != UC_ERR_OK) {
 102|                 goto error;
 103|             }
 104|         }
 105|     }
 106| 
 107|     if (m_size > 0 && !do_delete) {
 108|         if (!prealloc) {
 109|             if (uc_mem_map(uc, address, m_size, perms) != UC_ERR_OK) {
 110|                 goto error;
 111|             }
 112|             if (uc_mem_write(uc, address, backup + l_size, m_size) !=
 113|                 UC_ERR_OK) {
 114|                 goto error;
 115|             }
 116|         } else {
 117|             if (uc_mem_map_ptr(uc, address, m_size, perms, backup + l_size) !=
 118|                 UC_ERR_OK) {
 119|                 goto error;
 120|             }
 121|         }
 122|     }
 123| 
 124|     if (r_size > 0) {
 125|         if (!prealloc) {
 126|             if (uc_mem_map(uc, chunk_end, r_size, perms) != UC_ERR_OK) {
 127|                 goto error;
 128|             }
 129|             if (uc_mem_write(uc, chunk_end, backup + l_size + m_size, r_size) !=
 130|                 UC_ERR_OK) {
 131|                 goto error;
 132|             }
 133|         } else {
 134|             if (uc_mem_map_ptr(uc, chunk_end, r_size, perms,
 135|                                backup + l_size + m_size) != UC_ERR_OK) {
 136|                 goto error;
 137|             }
 138|         }
 139|     }
 140| 
 141|     if (!prealloc) {
 142|         free(backup);
 143|     }
 144|     return true;
 145| 
 146| error:
 147|     if (!prealloc) {
 148|         free(backup);
 149|     }
 150|     return false;
 151| }
