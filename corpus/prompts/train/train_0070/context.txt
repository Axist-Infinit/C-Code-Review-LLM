 545| 		fail_secure(zone, keyname);
 546| 	}
 547| 
 548|  done:
 549| 	if (!ISC_LIST_EMPTY(diff.tuples)) {
 550| 		/* Write changes to journal file. */
 551| 		CHECK(update_soa_serial(kfetch->db, ver, &diff, mctx,
 552| 					zone->updatemethod));
 553| 		CHECK(zone_journal(zone, &diff, NULL, "keyfetch_done"));
 554| 		commit = true;
 555| 
 556| 		DNS_ZONE_SETFLAG(zone, DNS_ZONEFLG_LOADED);
 557| 		zone_needdump(zone, 30);
 558| 	} else if (result == ISC_R_NOMORE) {
 559| 		/*
 560| 		 * If "updatekey" was true for all keys found in the DNSKEY
 561| 		 * response and the previous update of those keys happened
 562| 		 * during the same second (only possible if a key refresh was
 563| 		 * externally triggered), it may happen that all relevant
 564| 		 * update_one_rr() calls will return ISC_R_SUCCESS, but
 565| 		 * diff.tuples will remain empty.  Reset result to
 566| 		 * ISC_R_SUCCESS to prevent a bogus warning from being logged.
 567| 		 */
 568| 		result = ISC_R_SUCCESS;
 569| 	}
 570| 
 571|  failure:
 572| 	if (result != ISC_R_SUCCESS) {
 573| 		dnssec_log(zone, ISC_LOG_ERROR,
 574| 			   "error during managed-keys processing (%s): "
 575| 			   "DNSSEC validation may be at risk",
 576| 			   isc_result_totext(result));
 577| 	}
 578| 	dns_diff_clear(&diff);
 579| 	if (ver != NULL) {
 580| 		dns_db_closeversion(kfetch->db, &ver, commit);
 581| 	}
 582| 
 583|  cleanup:
 584| 	dns_db_detach(&kfetch->db);
 585| 
 586| 	INSIST(zone->irefs > 0);
 587| 	zone->irefs--;
 588| 	kfetch->zone = NULL;
 589| 
 590| 	if (dns_rdataset_isassociated(&kfetch->keydataset)) {
 591| 		dns_rdataset_disassociate(&kfetch->keydataset);
 592| 	}
 593| 	if (dns_rdataset_isassociated(&kfetch->dnskeyset)) {
 594| 		dns_rdataset_disassociate(&kfetch->dnskeyset);
 595| 	}
 596| 	if (dns_rdataset_isassociated(&kfetch->dnskeysigset)) {
 597| 		dns_rdataset_disassociate(&kfetch->dnskeysigset);
 598| 	}
 599| 
 600| 	dns_name_free(keyname, mctx);
 601| 	isc_mem_put(mctx, kfetch, sizeof(dns_keyfetch_t));
 602| 	isc_mem_detach(&mctx);
 603| 
 604| 	if (secroots != NULL) {
 605| 		dns_keytable_detach(&secroots);
 606| 	}
 607| 
 608| 	free_needed = exit_check(zone);
 609| 	UNLOCK_ZONE(zone);
 610| 	if (free_needed) {
 611| 		zone_free(zone);
 612| 	}
 613| 
 614| 	INSIST(ver == NULL);
 615| }
