  14|   // Make sure data dictionary is loaded.

  39|     // read socket handle number from stdin, i.e. the anonymous pipe

  41|     if (ReadFile(hStdIn, buf, sizeof(buf), &bytesRead, NULL))

  43|       // make sure buffer is zero terminated
