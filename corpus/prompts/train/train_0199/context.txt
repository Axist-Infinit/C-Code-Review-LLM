   5| 	struct debug_store *ds = cpuc->ds;

   7| 	void *base, *at, *top;
   8| 	short counts[INTEL_PMC_IDX_FIXED + MAX_FIXED_PEBS_EVENTS] = {};

  10| 	int bit, i, size;
  11| 	u64 mask;

  13| 	if (!x86_pmu.pebs_active)
  14| 		return;

  18|

  25| 		size = INTEL_PMC_IDX_FIXED + x86_pmu.num_counters_fixed;
  26| 	}
  27| 
  28| 	if (unlikely(base >= top)) {
  29| 		intel_pmu_pebs_event_update_no_drain(cpuc, size);
  30| 		return;
  31| 	}

  34| 		struct pebs_record_nhm *p = at;

  38| 		pebs_status &= mask;

  42| 			for_each_set_bit(bit, (unsigned long *)&pebs_status, size)
  43| 				counts[bit]++;

  45| 			continue;
  46| 		}

  49| 		 * On some CPUs the PEBS status can be zero when PEBS is
  50| 		 * racing with clearing of GLOBAL_STATUS.

  53| 		 * case when there is only a single active PEBS event
  54| 		 * we can assume it's for that event.

  66| 		 * The PEBS hardware does not deal well with the situation
  67| 		 * when events happen near to each other and multiple bits
  68| 		 * are set. But it should happen rarely.
  69| 		 *

  71| 		 * events, it doesn't impact PEBS record. The record will
  72| 		 * be handled normally. (slow path)
  73| 		 *

  75| 		 * records for the events can be collapsed into a single

  84| 		}

  86| 		counts[bit]++;
  87| 	}

  95| 			continue;
